async function loadRespo(name){ const r = await fetch(name + ".json"); return await r.json(); } async function buildMask(){ const names = ["REAL","BIL"]; const mask = []; for(const n of names){ const res = await loadRespo(n); mask.push({ pos: [res.RAUM.X, res.RAUM.Z], // RESPO → Raster refs: [res.NAME] }); } return mask; } buildMask().then(mask => render3x11(mask));
