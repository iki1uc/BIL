// ------------------------------------------------------ // FULL RESPO VIEWER (REAL + BIL) // ------------------------------------------------------ const RESPO_NAMES = ["REAL", "BIL"]; // JSON laden async function loadRespo(name){ try { const r = await fetch(name + ".json"); return await r.json(); } catch(e){ return { NAME:name, MISS:true }; } } // RESPO → Raster async function buildMask(){ const mask = []; for(const n of RESPO_NAMES){ const res = await loadRespo(n); if(res.MISS){ mask.push({ pos: [1,1], refs: [n + " MISSING"] }); continue; } mask.push({ pos: [res.RAUM.X, res.RAUM.Z], refs: [res.NAME] }); } return mask; } // Viewer starten buildMask().then(mask => render3x11(mask));
